This adds a great layer of "gambling" to the game. I’ve updated the logic so the game now tracks a Multiplier.

How the "Double Up" Logic Works:

  • The Multiplier: Starts at 1x. Every time you hit "Double Up," the multiplier doubles (2x, 4x, 8x...).
  • The Display: The result text automatically calculates the new requirement (e.g., if the multiplier is 2x, landing on "Remove 1 Clothing" will display "REMOVE 2 CLOTHING").
  • The Timer: The 1-minute timer is now dynamic. If you have a 2x multiplier and land on a timed action, the timer will automatically set itself to 120 seconds (2 minutes).
  • Reset: After a successful action and timer completion, the multiplier resets to 1x for the next round.

        const canvas = document.getElementById('wheel');

        const ctx = canvas.getContext('2d');

        const spinBtn = document.getElementById('spin-btn');

        const timerBtn = document.getElementById('timer-btn');

        const resultDisplay = document.getElementById('result-display');

        const timerDisplay = document.getElementById('timer-display');

        const wheelContainer = document.getElementById('wheel-container');

        const statusDisplay = document.getElementById('status-bar');


        const actions = [

            { text: "Remove 1 Clothing", type: "item", val: 1 },

            { text: "Remove 1 Clothing", type: "item", val: 1 },

            { text: "Remove 1 Clothing", type: "item", val: 1 },

            { text: "Remove 1 of HIS", type: "item", val: 1 },

            { text: "Kiss", type: "time", val: 60 },

            { text: "Nipple Play", type: "time", val: 60 },

            { text: "Suck", type: "time", val: 60 },

            { text: "Rim", type: "time", val: 60 },

            { text: "Finger", type: "time", val: 60 },

            { text: "DOUBLE UP", type: "special", val: 0 }

        ];


        let currentRotation = 0;

        let multiplier = 1;

        let timerInterval;

        let currentTargetSeconds = 60;


        function drawWheel() {

            canvas.width = 800; canvas.height = 800;

            const slice = (Math.PI * 2) / actions.length;

            actions.forEach((act, i) => {

                const angle = i * slice;

                ctx.beginPath();

                ctx.fillStyle = (i % 2 === 0) ? '#111' : '#222';

                if(act.type === 'special') ctx.fillStyle = '#330000';

                ctx.moveTo(400, 400);

                ctx.arc(400, 400, 400, angle, angle + slice);

                ctx.fill();

                ctx.strokeStyle = "#444"; ctx.lineWidth = 2; ctx.stroke();

                ctx.save();

                ctx.translate(400, 400);

                ctx.rotate(angle + slice / 2);

                ctx.textAlign = "right"; ctx.fillStyle = "#fff";

                ctx.font = "bold 26px monospace";

                ctx.fillText(act.text, 380, 10);

                ctx.restore();

            });

        }


        spinBtn.addEventListener('click', () => {

            const spinAdd = Math.floor(Math.random() * 360) + 3600; 

            currentRotation += spinAdd;

            canvas.style.transform = `rotate(${currentRotation}deg)`;

            resultDisplay.innerText = "SPINNING...";

            spinBtn.classList.add('hidden');


            setTimeout(() => {

                const actualDegree = currentRotation % 360;

                const index = Math.floor((360 - (actualDegree % 360)) / (360 / actions.length)) % actions.length;

                const result = actions[index];


                if (result.type === "special") {

                    multiplier *= 2;

                    statusDisplay.innerText = `MULTIPLIER: ${multiplier}X`;

                    resultDisplay.innerText = "DOUBLE UP! SPIN AGAIN!";

                    spinBtn.classList.remove('hidden');

                } else {

                    if (result.type === "item") {

                        const count = result.val * multiplier;

                        resultDisplay.innerText = `${result.text.replace('1', count)}`;

                        spinBtn.classList.remove('hidden');

                        multiplier = 1; // Reset after action

                    } else if (result.type === "time") {

                        currentTargetSeconds = result.val * multiplier;

                        resultDisplay.innerText = `${result.text} for ${currentTargetSeconds/60}m`;

                        timerBtn.innerText = `START ${currentTargetSeconds}s TIMER`;

                        timerBtn.classList.remove('hidden');

                    }

                }

            }, 5000);

        });


        timerBtn.addEventListener('click', () => {

            timerBtn.classList.add('hidden');

            wheelContainer.style.opacity = "0.2";

            timerDisplay.style.display = "block";

            startCountdown(currentTargetSeconds);

        });


        function startCountdown(seconds) {

            let timeLeft = seconds;

            timerDisplay.innerText = timeLeft + "s";

            timerInterval = setInterval(() => {

                timeLeft--;

                timerDisplay.innerText = timeLeft + "s";

                if (timeLeft <= 0) {

                    clearInterval(timerInterval);

                    timerDisplay.innerText = "DONE!";

                    setTimeout(() => {

                        multiplier = 1;

                        statusDisplay.innerText = `MULTIPLIER: 1X`;

                        resetUI();

                    }, 2000);

                }

            }, 1000);

        }


        function resetUI() {

            timerDisplay.style.display = "none";

            wheelContainer.style.opacity = "1";

            spinBtn.classList.remove('hidden');

            resultDisplay.innerText = "READY";

        }


        drawWheel();
